list(APPEND DIXTER_VSYNTH_INCLUDES  ${DIXTER_INCLUDE_DIR} ${DIXTER_INCLUDE_DIR}/SpeechSynthesizer)
list(APPEND DIXTER_VSYNTH_LIBRARIES ${DIXTER_VSYNTH})

set(VSYNTH_TARGET DixterVSynth)
set(VSYNTH_TARGET_SOURCES DixterVSynth.cpp)

add_target_module(${VSYNTH_TARGET} DIXTER_VSYNTH "${VSYNTH_TARGET_SOURCES}"
                  "${DIXTER_VSYNTH_INCLUDES}" "${DIXTER_VSYNTH_LIBRARIES}")

add_custom_command(
        TARGET ${VSYNTH_TARGET} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy
        $<TARGET_FILE:${VSYNTH_TARGET}>
        ${EXECUTABLE_OUTPUT_PATH}
)
